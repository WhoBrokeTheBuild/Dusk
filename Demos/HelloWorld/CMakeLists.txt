
ADD_EXECUTABLE(
    HelloWorld
    src/Main.cpp
)

TARGET_LINK_LIBRARIES(
    HelloWorld
    Dusk
)

LIST(INSERT _ASSET_PATH 0 "${CMAKE_CURRENT_SOURCE_DIR}/Assets")
SET(DUSK_ASSET_PATH "")
FOREACH(path ${_ASSET_PATH})
    FILE(RELATIVE_PATH relpath "${CMAKE_CURRENT_SOURCE_DIR}" "${path}")
    LIST(APPEND DUSK_ASSET_PATH "${relpath}")
ENDFOREACH()

STRING(REPLACE ";" ":" DUSK_ASSET_PATH "${DUSK_ASSET_PATH}")

TARGET_COMPILE_DEFINITIONS(
    HelloWorld 
    PRIVATE
        DUSK_ASSET_PATH="${DUSK_ASSET_PATH}"
)

ADD_CUSTOM_TARGET(
    run-HelloWorld
    COMMAND HelloWorld
    DEPENDS HelloWorld
    WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
)
