SET(_ENGINE "Dusk")

SET(_ENGINE_CONFIG_IN  "${CMAKE_CURRENT_SOURCE_DIR}/include/dusk/Config.hpp.in")
SET(_ENGINE_CONFIG_OUT "${CMAKE_CURRENT_BINARY_DIR}/include/dusk/Config.hpp")

CONFIGURE_FILE(${_ENGINE_CONFIG_IN} ${_ENGINE_CONFIG_OUT})

FILE(GLOB_RECURSE 
    _ENGINE_INCLUDES 
    "${CMAKE_CURRENT_SOURCE_DIR}/include/*.hpp"
)

FILE(GLOB_RECURSE 
    _ENGINE_SOURCES  
    "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/src/*.c"
    "${CMAKE_CURRENT_SOURCE_DIR}/src/*.hpp"
    "${CMAKE_CURRENT_SOURCE_DIR}/src/*.h"
)

IF(WIN32)
    ADD_LIBRARY(
        ${_ENGINE}
        ${_ENGINE_INCLUDES}
        ${_ENGINE_SOURCES}
        ${_ENGINE_CONFIG_OUT}
    )
ELSE()
    ADD_LIBRARY(
        ${_ENGINE} SHARED 
        ${_ENGINE_INCLUDES}
        ${_ENGINE_SOURCES}
        ${_ENGINE_CONFIG_OUT}
    )
ENDIF()

TARGET_INCLUDE_DIRECTORIES(
    ${_ENGINE}
    PUBLIC
        $<INSTALL_INTERFACE:include>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/include>
        $<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}/include>
    PRIVATE
        "${CMAKE_CURRENT_SOURCE_DIR}/include"
        "${CMAKE_CURRENT_SOURCE_DIR}/src"
        "${CMAKE_CURRENT_BINARY_DIR}/include"
)

TARGET_COMPILE_FEATURES(
    ${_ENGINE} 
    PRIVATE
        cxx_std_17
        cxx_auto_type
)

TARGET_COMPILE_OPTIONS(
    ${_ENGINE} 
    PRIVATE
        $<$<CXX_COMPILER_ID:GNU>:-Wall>
        $<$<CXX_COMPILER_ID:Clang>:-Wall>
)

TARGET_LINK_LIBRARIES(
    ${_ENGINE}
    GLM::GLM
    JSON::JSON
    SDL2::SDL2
)

IF(MSVC)
    SET_TARGET_PROPERTIES(
        ${_ENGINE} PROPERTIES
        LINK_FLAGS "/NODEFAULTLIB:LIBCMT /NODEFAULTLIB:MSVCRT" 
    )
ENDIF()
